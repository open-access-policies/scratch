% Define table colors
\definecolor{stripecolor}{HTML}{f2f2f2}
\definecolor{headcolor}{HTML}{e6e6e6}
\definecolor{bordercolor}{HTML}{999999}

% Table styling
\renewcommand{\arraystretch}{1.3}
\setlength{\tabcolsep}{8pt}

% Make table headers bold and centered
\renewcommand{\theadfont}{\bfseries}
\renewcommand{\theadalign}{cc}

% Add borders and improved styling to tables
\AtBeginEnvironment{tabular}{%
  \rowcolors{2}{white}{stripecolor}
  \arrayrulecolor{bordercolor}
}
\AtBeginEnvironment{longtable}{%
  \rowcolors{2}{white}{stripecolor}
  \arrayrulecolor{bordercolor}
}

% Improve table captions
\captionsetup[table]{%
  font=small,
  labelfont=bf,
  format=plain,
  labelsep=period,
  justification=raggedright,
  singlelinecheck=off
}

% Custom command for nicer table headers
\newcommand{\headrow}{\rowcolor{headcolor}}

% Define column types for better text wrapping
\newcolumntype{L}[1]{>{\raggedright\arraybackslash}p{#1}}
\newcolumntype{C}[1]{>{\centering\arraybackslash}p{#1}}
\newcolumntype{R}[1]{>{\raggedleft\arraybackslash}p{#1}}
\newcolumntype{Y}{>{\centering\arraybackslash}X}
